Hover Lines

The HoverLines component is an interactive vertical navigation menu with animated headings and lines that dynamically resize based on the user's mouse movement. It uses Framer Motion for smooth spring animations, creating an engaging visual effect.

import { ClassValue, clsx } from "clsx";
import { twMerge } from "tailwind-merge";


export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}

components/ui/hover-lines.tsx

1"use client";
2

3import { cn } from "@/lib/utils";
4import { motion, useAnimationControls, useMotionValue } from "framer-motion";
5import React, { useRef, useState } from "react";
6

7type LineAnimationProps = {
8  width: number;
9};
10

11export default function HoverLines({ headings }: { headings: string[] }) {
12  const style = `h-1 w-4 bg-primary`;
13

14  const smallLineRef = useRef<(HTMLDivElement | null)[]>([]);
15

16  const bigLineControls = useAnimationControls();
17  const headingControls = useAnimationControls();
18

19  const mouseY = useMotionValue(0);
20

21  const [lineAnimationProps, setLineAnimationProps] = useState<
22    LineAnimationProps[]
23  >([]);
24

25  function handleMouseMove(
26    event: React.MouseEvent<HTMLDivElement, MouseEvent>
27  ) {
28    mouseY.set(event.clientY);
29

30    const newAnimationProps = smallLineRef.current.map((ele) => {
31      if (!ele) return { width: 16 };
32

33      const { top } = ele.getBoundingClientRect();
34      const diff = Math.abs(mouseY.get() - top);
35

36      const width = Math.max((1 - Math.min(diff / 150, 1)) * 10 * 16, 16);
37

38      return {
39        width,
40      };
41    });
42

43    setLineAnimationProps(newAnimationProps);
44  }
45

46  const handleMouseEnter = () => {
47    bigLineControls.start("hover");
48    headingControls.start("hover");
49  };
50

51  const handleMouseLeave = () => {
52    bigLineControls.start("noHover");
53    headingControls.start("noHover");
54    const resetAnimationProps = smallLineRef.current.map(() => ({
55      width: 16,
56      transition: {
57        type: "spring",
58      },
59    }));
60

61    setLineAnimationProps(resetAnimationProps);
62  };
63

64  return (
65    <div
66      className="fixed right-0 top-0 h-screen flex flex-col justify-around w-[14rem] overflow-hidden"
67      onMouseMove={handleMouseMove}
68      onMouseEnter={handleMouseEnter}
69      onMouseLeave={handleMouseLeave}
70    >
71      {headings.map((ele, index1) => {
72        return (
73          <div
74            key={ele}
75            className="flex flex-col justify-around h-full items-end"
76          >
77            <motion.div
78              className={cn(style, "w-8")}
79              animate={bigLineControls}
80              variants={{
81                hover: { width: "12rem" },
82                noHover: { width: "2rem" },
83              }}
84              transition={{
85                type: "spring",
86              }}
87            >
88              <motion.p
89                variants={{
90                  noHover: { x: "3rem" },
91                  hover: { x: "0.4rem" },
92                }}
93                className="translate-x-8 w-48"
94                animate={headingControls}
95              >
96                {ele}
97              </motion.p>
98            </motion.div>
99            {Array.from({ length: 4 }, (_, index2) => index2 + 1).map(
100              (ele, index2) => {
101                const currentLineIndex = index1 * 4 + index2;
102

103                return (
104                  <motion.div
105                    className={cn(style)}
106                    key={index2}
107                    ref={(ele: HTMLDivElement | null) => {
108                      smallLineRef.current[currentLineIndex] = ele;
109                    }}
110                    animate={lineAnimationProps[currentLineIndex]}
111                    transition={{
112                      type: "spring",
113                      damping: 50,
114                      stiffness: 1000,
115                      // duration:0.1,
116                    }}
117                  />
118                );
119              }
120            )}
121          </div>
122        );
123      })}
124    </div>
125  );
126}
